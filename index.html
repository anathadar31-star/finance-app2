function displayData(rows) { const tbody = document.getElementById("tableBody"); tbody.innerHTML = ""; let total = 0; // פונקציה להמרת תאריך אקסל function excelDateToJSDate(serial) { const utc_days = Math.floor(serial - 25569); const utc_value = utc_days * 86400; const date_info = new Date(utc_value * 1000); return date_info.toLocaleDateString("he-IL"); } for (let i = 1; i < rows.length; i++) { const row = rows[i]; if (!row) continue; // דילוג על שורות כותרת if (row[0] === "תאריך" || row[1] === "הפעולה") continue; let dateRaw = row[0]; let date = ""; if (typeof dateRaw === "number") { date = excelDateToJSDate(dateRaw); } else { date = dateRaw || ""; } const desc = row[2] || ""; const debit = parseFloat(row[4]) || 0; const credit = parseFloat(row[5]) || 0; let amount = 0; if (credit > 0) { amount = credit; } else if (debit > 0) { amount = -debit; } if (amount !== 0) total += amount; tbody.innerHTML += ` ${date} ${desc} ${amount !== 0 ? amount.toLocaleString("he-IL") : ""} `; } document.getElementById("total").innerText = "מאזן כולל: ₪ " + total.toLocaleString("he-IL"); document.getElementById("status").innerText = "הטעינה הסתיימה"; }
